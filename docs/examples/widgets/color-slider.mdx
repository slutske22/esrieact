# ColorSlider

The `HistogramRangeSlider` widget is a slider that allows the user to select a range of values from a histogram.

## Simple example

The following example shows how you can capture `events` on a `HistogramRangeSlider` widget, and use the captures values to manipulate the `FeatureLayerView` of a `FeatureLayer`.

```tsx live
function Example() {
  const [layerRef, setLayerRef] = useState<__esri.FeatureLayer>(null);
  const [rendererResponse, setRendererResponse] =
    useState<__esri.RendererResult>(null);

  useEffect(() => {
    if (layerRef) {
      const createRenderer = async () => {
        const response = await createContinuousRenderer({
          layer: layerRef,
          field: "pop20",
          theme: "high-to-low",
        });
        setRendererResponse(response);
      };
      createRenderer();
    }
  }, [layerRef]);

  return (
    <MapView
      ViewProperties={{ extent: OAHU_EXTENT_2 }}
      MapProperties={{ basemap: "topo-vector" }}
      style={{ height: "300px", width: "100%" }}
    >
      <FeatureLayer
        url={HI_CENSUS_2020_FL_URL}
        events={{ "layerview-create": (e) => setLayerRef(e.layerView.layer) }}
        renderer={rendererResponse?.renderer}
      />

      <Legend position="top-right" />

      <h4 className="mt-2! mb-0!">Population 2020</h4>
      <div
        id="color-slider-container"
        style={{ height: "300px", width: "100%" }}
      >
        {rendererResponse && (
          <ColorSlider
            container="color-slider-container"
            stops={rendererResponse.visualVariable.stops}
            min={rendererResponse.statistics.min as number}
            max={rendererResponse.statistics.max as number}
          />
        )}
      </div>
    </MapView>
  );
}
```
